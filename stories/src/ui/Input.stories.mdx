## With Ref

<Canvas>
  <Story name="With Ref">
    <Fragment>
      {(() => {
        const ref = createRef()
        useEffect(() => {
          const element = ref.current
          if (!element) {
            return undefined
          }
          function onInput(e) {
            action(`Value changed! ${e.target.value}`)(e)
          }
          element.addEventListener('input', onInput)
          return () => {
            element.removeEventListener('input', onInput)
          }
        }, [ref])
        return <Input ref={ref} />
      })()}
    </Fragment>
  </Story>
</Canvas>
