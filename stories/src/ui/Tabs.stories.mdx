import { Meta, Story, Canvas } from '@storybook/addon-docs'
import { Tabs, Tab } from '@amsterdam/asc-ui'

export const PaddedContent = ({ children }) => (
  <div style={{ paddingTop: '20px' }}>{children}</div>
)

export const tabOptions = { First: 'one', Second: 'two', Last: 'three' }

export const TabsStory = (args) => (
  <Tabs label="An example of tabs" {...args}>
    {Object.entries(tabOptions).map(([key, value]) => (
      <Tab id={value} label={key}>
        <PaddedContent>Contents of the {key.toLowerCase()} tab.</PaddedContent>
      </Tab>
    ))}
  </Tabs>
)

<Meta
  title="UI/Tabs"
  component={Tabs}
  argTypes={{
    activeTab: {
      options: tabOptions,
      control: { type: 'select' },
    },
    initialTab: {
      options: tabOptions,
      control: { type: 'select' },
    },
  }}
/>

# Tabs

Tabs organise and allow navigation between groups of content that are related and at the same level of hierarchy.

[Design System](https://designsystem.amsterdam.nl/7awj1hc9f/p/15737c-tab-navigation)
