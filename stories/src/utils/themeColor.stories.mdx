# themeColor

A function to retrieve a color from a theme

## Primary colors

<Canvas>
  <Story name="Primary colors">
    <Fragment>
      {(() => {
        const Primary = styled.div`
          color: ${themeColor('tint', 'level1')};
          background-color: ${themeColor('primary')};
        `
        const PrimaryDark = styled.div`
          color: ${themeColor('tint', 'level1')};
          background-color: ${themeColor('primary', 'dark')};
        `
        const Secondary = styled.div`
          color: ${themeColor('tint', 'level1')};
          background-color: ${themeColor('secondary')};
        `
        return (
          <Fragment>
            <Primary>themeColor(&apos;primary&apos;)</Primary>
            <PrimaryDark>
              themeColor(&apos;primary&apos;, &apos;dark&apos;)
            </PrimaryDark>
            <Secondary>themeColor(&apos;secondary&apos;)</Secondary>
          </Fragment>
        )
      })()}
    </Fragment>
  </Story>
</Canvas>
