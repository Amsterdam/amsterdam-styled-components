import { Meta, Story, Canvas } from '@storybook/addon-docs'
import { Icon, ThemeProvider, GlobalStyle } from '@amsterdam/asc-ui'
import { Button, ascBetaTheme } from '@amsterdam/asc-ui-beta'
import styled from 'styled-components'
import { Download } from '@amsterdam/asc-assets'

<Meta
  title="Beta/Button"
  component={Button}
  argTypes={{
    small: {
      control: { type: 'boolean' },
    },
    gutterBottom: {
      control: { type: 'number' },
    },
  }}
/>

# Button

## Default / Primary variant

export const ButtonBar = styled.div`
  display: flex;
  flex-wrap: wrap;
  & > * {
    margin-right: 5px;
  }
`

export const StyledIcon = styled(Icon)`
  margin-right: 16px;
`

<Canvas>
  <Story
    name="Default / Primary variant"
    args={{ small: false, gutterBottom: 0 }}
  >
    {({ small, gutterBottom }) => (
      <ThemeProvider theme={ascBetaTheme}>
        <GlobalStyle />
        <ButtonBar>
          <Button small={small} gutterBottom={gutterBottom}>
            Normal
          </Button>
          <Button small={small} gutterBottom={gutterBottom} disabled>
            Disabled
          </Button>
          <Button small={small} gutterBottom={gutterBottom}>
            <StyledIcon size={20}>
              <Download />
            </StyledIcon>
            With icon
          </Button>
        </ButtonBar>
      </ThemeProvider>
    )}
  </Story>
</Canvas>
